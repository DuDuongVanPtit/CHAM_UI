    CHAM TOEIC Navigation System

    Desktop Features

    • • Full horizontal menu (1200px container)
    • • Dropdown menus with hover effects
    • • Red underline animations
    • • Sticky header with scroll effects
    • • 80px height (70px when scrolled)

    Tablet Behavior

    • • Condensed menu layout
    • • "Thêm" dropdown for extra items
    • • Responsive spacing adjustments
    • • Optimized touch interactions
    • • Maintained functionality

    Mobile Experience

    • • Hamburger menu button
    • • Full-screen overlay (slide from right)
    • • Collapsible dropdown sections
    • • 64px header height (60px scrolled)
    • • Touch-optimized interactions

    React Component Implementation

    // Header.jsx
import React, { useState, useEffect } from 'react';

const Header = () => {
  const [isMobileMenuOpen, setIsMobileMenuOpen] = useState(false);
  const [isScrolled, setIsScrolled] = useState(false);
  const [activeDropdown, setActiveDropdown] = useState(null);

  useEffect(() => {
    const handleScroll = () => {
      setIsScrolled(window.scrollY > 50);
    };
    window.addEventListener('scroll', handleScroll);
    return () => window.removeEventListener('scroll', handleScroll);
  }, []);

  const navigationItems = [
    {
      label: 'Về chúng tôi',
      hasDropdown: true,
      items: ['Giới thiệu', 'Tầm nhìn & Sứ mệnh', 'Thành tích', 'Chứng nhận']
    },
    {
      label: 'Đội ngũ đào tạo',
      hasDropdown: true,
      tabletHidden: true,
      items: ['Giáo viên bản ngữ', 'Giáo viên Việt Nam', 'Chuyên gia TOEIC']
    },
    // ... more items
  ];

  return (
    <header className={`header ${isScrolled ? 'scrolled' : ''}`}>
      <div className="header-container">
        <Logo />
        <Navigation items={navigationItems} />
        <CTAButton />
        <MobileMenuButton onClick={() => setIsMobileMenuOpen(!isMobileMenuOpen)} />
      </div>
      <MobileMenu isOpen={isMobileMenuOpen} items={navigationItems} />
    </header>
  );
};

export default Header;
